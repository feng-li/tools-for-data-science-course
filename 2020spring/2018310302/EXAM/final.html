import requests
import re
import time
from bs4 import BeautifulSoup
from collections import OrderedDict

def get_news1():
    """
    page
    """
    newsData = OrderedDict()
    href = "http://jwc.cufe.edu.cn/tzgg.htm"
    html = requests.get(href)
    # Parsing html
    soup = BeautifulSoup(html.content, 'html.parser')
    div= soup.find('div', {"class": "listR_nr"})
    lis = div.findAll('li')
    for li in lis:
        titleinfo = li.find('a')
        title = titleinfo.get_text()
        a = re.compile(r'\u325b|\u2b07|\ufe0f|\ufffd|\u200b')
        title = a.sub('', title)
        # News url
        url = "http://jwc.cufe.edu.cn/"+titleinfo['href']
        # date
        date = li.find('span').get_text()
        newsData[title] = [date, url]
    return newsData

import smtplib
from email.mime.text import MIMEText
from email.utils import formataddr

send_user = "************"       # 
send_pwd = "*************"        # 
rec_user = "***********"    #

def mail(data):
    ret = True
    #  plain  utf-8 
    msg = MIMEText(data, "plain", "utf-8")
    msg["From"] = formataddr(["-Li_nine", send_user])  # 
    msg["To"] = formataddr(["RecSMTP", rec_user])  # 
    msg["Subject"] = ""  # 

    server = smtplib.SMTP_SSL("smtp.qq.com", 465)   # SMTP
    server.login(send_user, send_pwd)  # 
    server.sendmail(send_user, rec_user, msg.as_string()) # 
    server.quit()  # 

    return ret
