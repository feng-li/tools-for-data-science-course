用Python实现校园通知更新提醒¶

简介¶

学校总是有很多的通知，分别在不同的网站上发布，如学校主页、教务处、图书馆以及学院的网站等，一个个查看过于麻烦，利用python写一个中央财经大学的通知爬虫，设计一个本校消息的自动通知程序

实现¶

  • 使用requests库获取到目标url(各个网站的’更多通知’页)的源代码，然后利用Python的BeautifulSoup库进行信息的筛选，提取到通知相关信息后，与当前时间相对比，如果为今天则加入一个OrderedDict中，保存为新通知。
  • 若检测到有新的通知，则利用smtplib库向指定邮箱发送提醒信息。

import requests
import re
import time
from bs4 import BeautifulSoup
from collections import OrderedDict

def get_news1():
    """
    获取教务处网址上的page页的新闻，返回一个通知字典
    """
    newsData = OrderedDict()
    href = "http://jwc.cufe.edu.cn/tzgg.htm"
    html = requests.get(href)
    # Parsing html
    soup = BeautifulSoup(html.content, 'html.parser')
    div= soup.find('div', {"class": "listR_nr"})
    lis = div.findAll('li')
    for li in lis:
        titleinfo = li.find('a')
        title = titleinfo.get_text()
        a = re.compile(r'\u325b|\u2b07|\ufe0f|\ufffd|\u200b')
        title = a.sub('', title)
        # News url
        url = "http://jwc.cufe.edu.cn/"+titleinfo['href']
        # date
        date = li.find('span').get_text()
        newsData[title] = [date, url]
    return newsData

import smtplib
from email.mime.text import MIMEText
from email.utils import formataddr

send_user = "************"       # 发件人的邮箱账号
send_pwd = "*************"        # 发件人邮箱的密码
rec_user = "***********"    #收件人邮箱

def mail(data):
    ret = True
    # 三个参数：第一个为文本内容，第二个 plain 设置文本格式，第三个 utf-8 设置编码
    msg = MIMEText(data, "plain", "utf-8")
    msg["From"] = formataddr(["自动提醒教务处通知-Li_nine", send_user])  # 括号中对应发件人邮箱昵称、发件人邮箱账号
    msg["To"] = formataddr(["RecSMTP", rec_user])  # 括号中对应收件人邮箱昵称、收件人邮箱账号
    msg["Subject"] = "今日教务处通知提醒"  # 邮件的主题或标题

    server = smtplib.SMTP_SSL("smtp.qq.com", 465)   # 括号中对应的是发件人邮箱中的SMTP服务器，端口
    server.login(send_user, send_pwd)  # 括号中对应的是发件人邮箱账号和密码
    server.sendmail(send_user, rec_user, msg.as_string()) # 括号中对应的是发件人邮箱账号、收件人邮箱账号、发送邮件
    server.quit()  # 关闭连接

    return ret
